#### val();

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * val();
 *
 * 仅可以作用于 form 表单的标签中使用
 *
 * 在同父级下，有且仅获取一个DOM元素的 value 值
 * 无论是否选择单个元素，或父级元素( 返回第一个元素的 value 值 )，都只返回单个元素的 value 值
 */


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// get value

console.log($('input[type=checkbox]').val());           // three


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// set value

$('input[type=checkbox]').val(function (index, currentValue) {

    return currentValue + index;
});


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 仅改变了 input 输入框中的 value 值, 并未改变行间样式 value 的属性值

$('input').eq(0).val('hello world');


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### serialize

``` javascript


/**
 * serialize();
 *
 * serialize: 串联
 *
 * 获取表单中所有的信息，并返回以 '&' 连接的字符串 =>'key:value&key:value&key:value'
 */

$('form').serialize();                                  // 'star=a&easy=alpha&des='


```
